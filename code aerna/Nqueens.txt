#include <stdio.h>
#include <stdlib.h>

int n, board[10];

int isSafe(int row, int col) {
    for (int i = 0; i < row; i++) {
        if (board[i] == col || abs(board[i] - col) == abs(i - row))
            return 0;
    }
    return 1;
}

int solve(int row) {
    if (row == n)
        return 1;  // one solution found

    for (int col = 0; col < n; col++) {
        if (isSafe(row, col)) {
            board[row] = col;
            if (solve(row + 1))
                return 1;  // stop after one valid solution
        }
    }
    return 0;
}

int main() {
    printf("Enter number of queens: ");
    scanf("%d", &n);

    if (solve(0)) {
        printf("\nOne possible solution:\n");
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (board[i] == j)
                    printf("Q ");
                else
                    printf("- ");
            }
            printf("\n");
        }
    } else {
        printf("No solution exists for %d queens.\n", n);
    }

    return 0;
}
